\makeatletter
% provide access to internal counters from neuralnetwork.sty
\newcommand{\currentlayerindex}{\nn@get{thislayerindex}}
\newcommand{\currentlayercount}{\nn@get{thislayercount}}
\makeatother

\newcommand{\drawinputs}{%
\foreach \i in {1,...,\currentlayercount}
  \draw[link] ($(L\currentlayerindex-\i) - (1,0)$) node[anchor=east] {$x_\i$}
           -- (L\currentlayerindex-\i);
}

\newcommand{\drawoutputs}{%
\foreach \i in {1,...,\currentlayercount}
  \draw[link] (L\currentlayerindex-\i) -- +(1,0) node[anchor=west] {$y_\i$};
}

\newcommand{\sumlabel}[2]{$\Sigma$}

% add subcaptions to group plots
\newcommand{\groupcaption}[3][]{%
  \node[anchor=north,align=center,text width=1cm,#1] at (#2.south) {\subcaption{#3}}}
